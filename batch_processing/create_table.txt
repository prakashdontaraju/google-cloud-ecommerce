CREATE TABLE IF NOT EXISTS batch_data (
	event_time timestamp
	, event_type text
	, product_id text
	, category_id text
	, category_code text
	, brand text
	, price float
	, user_id text
	, user_session text
	, hour int
	, category text
	, sub_category text
	, product text
	, product_details text
    , PRIMARY KEY (event_id))
